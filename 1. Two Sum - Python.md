class Solution:
    def twoSum(self, nums, target):
        """
        :type nums: List[int]
        :type target: int
        :rtype: List[int]
        """
        if len(nums) <= 1:
            return False
        dict = {}
        for i in range(len(nums)):
            num2 = target - nums[i]
            if num2 in dict:
                return [dict[num2], i]
            else:
                dict[nums[i]] = i
